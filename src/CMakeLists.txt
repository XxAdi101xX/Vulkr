project(src)
set(COMMON_FILES
    # Header Files
    common/helpers.h
    common/logger.h
    common/vulkan_common.h
    common/strings.h
    common/fence_pool.h
    common/semaphore_pool.h
    common/timer.h
    common/obj_loader.h
    common/debug_util.h
    common/gltf_loader.h
    # Source Files
    common/vulkan_common.cpp
    common/strings.cpp
    common/fence_pool.cpp
    common/semaphore_pool.cpp
    common/timer.cpp
    common/obj_loader.cpp
    common/debug_util.cpp
    common/gltf_loader.cpp
)

set(CORE_FILES
    # Header Files
    core/device.h
    core/instance.h
    core/physical_device.h
    core/queue.h
    core/swapchain.h
    core/image_view.h
    core/image.h
    core/render_pass.h
    core/pipeline.h
    core/pipeline_layout.h
    core/framebuffer.h
    core/command_pool.h
    core/command_buffer.h
    core/buffer.h
    core/descriptor_set_layout.h
    core/descriptor_pool.h
    core/descriptor_set.h
    core/sampler.h
    # Source Files
    core/device.cpp
    core/instance.cpp
    core/physical_device.cpp
    core/queue.cpp
    core/swapchain.cpp
    core/image_view.cpp
    core/image.cpp
    core/render_pass.cpp
    core/pipeline.cpp
    core/pipeline_layout.cpp
    core/framebuffer.cpp
    core/command_pool.cpp
    core/command_buffer.cpp
    core/buffer.cpp
    core/descriptor_set_layout.cpp
    core/descriptor_pool.cpp
    core/descriptor_set.cpp
    core/sampler.cpp
)

set(PLATFORM_FILES
    # Header Files
    platform/input_event.h
    platform/window.h
    platform/platform.h
    platform/application.h
    # Source Files
    platform/input_event.cpp
    platform/window.cpp
    platform/platform.cpp
    platform/application.cpp
)

set(RENDERING_FILES
    # Header Files
    rendering/subpass.h
    rendering/shader_module.h
    rendering/pipeline_state.h
    rendering/camera.h
    rendering/camera_controller.h
    # Source Files
    rendering/subpass.cpp
    rendering/shader_module.cpp
    rendering/pipeline_state.cpp
    rendering/camera.cpp
    rendering/camera_controller.cpp
)

source_group("common\\" FILES ${COMMON_FILES})
source_group("core\\" FILES ${CORE_FILES})
source_group("platform\\" FILES ${PLATFORM_FILES})
source_group("rendering\\" FILES ${RENDERING_FILES})

set(PROJECT_FILES
    ${PLATFORM_FILES}
    ${COMMON_FILES}
    ${CORE_FILES}
    ${RENDERING_FILES}
)

add_library(${PROJECT_NAME} STATIC ${PROJECT_FILES})

target_include_directories(${PROJECT_NAME} PUBLIC ${CMAKE_CURRENT_SOURCE_DIR})

# Link third party libraries
target_link_libraries(${PROJECT_NAME}
    volk
    glm
    spdlog
    imgui
    VulkanMemoryAllocator
    stb
    tinyobjloader
    tinygltf
)

# Link platform specific libraries
if(ANDROID)
    target_link_libraries(${PROJECT_NAME} log android native_app_glue)
else()
    if (NOT DIRECT_TO_DISPLAY)
        target_link_libraries(${PROJECT_NAME} glfw)
    endif()
endif()
